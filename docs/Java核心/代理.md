<h2>静态代理</h2>
<pre>
<code>
// * 委托类
// 是方法的实际执行者
public class Entruster {
    @Override
    public void method() {
        // do something...
    }
}
// * 代理类
// 将方法的调用代理到委托类上
public class Proxy extends Entruster{
    // * 委托对象
    // 真正做事的对象，可以称之为 realSubject。
    private Entrust entruster;
    public Proxy(Entruster entruster) {
        this.entruster = entruster;
    }
    // 代理类实现委托类夫接口的方法，这样代理类就拥有了和委托类相同的方法
    @Override
    public void method() {
        // ...
        // 执行委托对象的方法
        entruster.method();
        // ...
    }
}
</code>
</pre>
